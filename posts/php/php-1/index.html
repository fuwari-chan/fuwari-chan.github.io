HOW TO COMPILE PHP ON REDHAT ARM64

Pengertian PHP#

¬†¬†Pengertian singkat PHP saja karena kita akan fokus bagaimana cara mengkompilasi ,PHP (Hypertext Preprocessor) adalah bahasa pemrograman open-source yang umumnya digunakan untuk membangun aplikasi web dinamis dan interaktif. PHP dapat dijalankan pada server web dan dikombinasikan dengan HTML, CSS, dan JavaScript untuk membuat halaman web yang dinamis.

  1. Download Source

Sebagai contoh kita menggunakan php versi 7.4.6


wget https://www.php.net/distributions/php-7.4.6.tar.gz -O php-7.4.6.tar.xz
  1. Mengkonfigurasi

#!/bin/bash
U=${SUDO_USER:=${USER}}
if [[ -x `which realpath` ]]; then
	T=$(realpath ${BASH_SOURCE[0]:=${0}})
else
	T=${BASH_SOURCE[0]:=${0}}
fi
D=$(dirname ${T})
mkdir tmp
#apt-get install libfcgi-dev libfcgi0ldbl libjpeg62-turbo-dbg libmcrypt-dev libssl-dev libc-client2007e libc-client2007e-dev libxml2-dev libbz2-dev libcurl4-openssl-dev libjpeg-dev libpng12-dev libfreetype6-dev libkrb5-dev libpq-dev libxml2-dev libtidy-dev libmemcached-dev imagemagick-dev msgpack-dev

PHP_VERSION=7.4.6
wget https://www.php.net/distributions/php-${PHP_VERSION}.tar.gz -O php-${PHP_VERSION}.tar.xz
tar -xvf php-${PHP_VERSION}.tar.xz -C tmp
cd ${D}/tmp/php-${PHP_VERSION}/ext
#git clone https://github.com/krakjoe/apcu.git
#git clone https://github.com/krakjoe/apcu-bc.git
#git clone https://github.com/php-memcached-dev/php-memcached.git memcached
#git clone https://github.com/mkoppanen/imagick.git
#git clone https://github.com/msgpack/msgpack-php msgpack
cd ..
#rm configure
./buildconf --force
./configure --prefix=${D}/php-7.4.6 --with-config-file-path=${D}/php-7.4.6/conf --with-config-file-scan-dir=${D}/php-7.4.6/conf/php.ini.d --localstatedir=${D}/php-7.4.6 \
--config-cache --disable-cgi --enable-static=yes --enable-shared=yes --with-tsrm-pthreads --disable-phpdbg --enable-xdebug \
--with-msgpack --with-zlib-dir --with-freetype-dir --enable-mbstring --with-libxml-dir=${D}/php-7.4.6/usr --enable-soap \
--enable-calendar --with-curl --with-mcrypt --with-zlib --with-jpeg --without-pgsql --disable-rpath --enable-inline-optimization --with-bz2 --with-zlib --with-cli --with-xml --with-common --with-json --with-dev  --with-ctype --with-fileinfo --with-sqlite3 \
--enable-sockets --enable-sysvsem --enable-sysvshm --enable-sysvmsg --enable-pcntl --enable-mbregex --enable-exif --enable-bcmath --with-mhash --enable-zip --with-pcre-regex --enable-gd --with-external-gd \
--with-pdo-mysql --without-pdo-pgsql --with-mysqli --enable-embedded-mysqli --with-mysql-sock=/var/run/mysqld/mysqld.sock --with-jpeg-dir=${D}/php-7.4.6/usr --with-png-dir=${D}/php-7.4.6/usr --enable-gd-native-ttf --with-openssl \
--with-fpm-user=www-data --with-fpm-group=www-data --with-libdir=${D}/php-7.4.6/libs --enable-ftp --without-kerberos --without-imap --without-imap-ssl \
--enable-intl --with-pcre-jit --enable-apc --enable-apcu --with-curl --with-imagick --enable-memcached --enable-memcached-protocol --enable-memcached-msgpack --enable-memcached-msgpack \
